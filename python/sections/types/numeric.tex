\subsection{Numeric types}
Integers, Floats, complex numbers \\
e.g. \texttt{10, 54, 2.5, 3.0, 3+4j, 6-2j} \\
\textit{Operations:} \texttt{+, -, *, /, \%, //, **, abs()}
\begin{lstlisting}
# x and y are complex numbers
x = 3+4j
print (x + 2)  # 5+4j
print (x.real, x.imag) # 3.0, 4.0
print (x.conjugate()) # 3-4j
print (abs(x)) # 5
print (5/2, 5//2) # 2.5 2
print (5**2) # 25
print (abs(-5)) # 5
\end{lstlisting}
