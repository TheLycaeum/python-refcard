\subsection{Sets}
\texttt{set([1,2,3])}  or \texttt{\{1,2,3\}}

\textit{Operations:} \texttt{-, \&, |, \textasciicircum, len}
\begin{lstlisting}
u = {1,2}
v = {2,3}
w = {2}
u.union(v) # {1, 2, 3}
u | v # {1, 2, 3}
u & v # {2}
u.intersection(v) # {2}
u - v # {1}
u.isdisjoint(v) # False
u.issuperset(w) # True
\end{lstlisting}
