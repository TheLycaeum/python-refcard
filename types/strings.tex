\subsection{Strings}
\texttt{``Example''}, \texttt{'example'}, \texttt{""" \\
Multiline \\
string """}\\
\textit{Operations:} \texttt{+, *, len()}
\begin{lstlisting}
x = "hello there"
x.upper() # 'HELLO THERE'
x.split() # ['hello', 'there']
x.center(20) # '    hello there     '
x.endswith("e") # True
x.startswith("x") # False

t = "{} in 2016"
t.format("python", 2016) # 'python in 2016'
\end{lstlisting}
