\subsection{Dictionary}
\texttt{\{"name":"Turing", "age":30\}} or \\
\texttt{dict(name="Turing", age=30)} \\

\textit{Operations} \\
\texttt{[], del}

\begin{lstlisting}
d = dict(x=1, y=2)
list(d) # ['y', 'x']
list(d.items()) # [('y', 2), ('x', 1)]
list(d.values()) # [2, 1]
'x' in d # True
'a' in d # False
d['x'] # 1
d['t'] # raises KeyError
d.get('t', "Foo") # 'Foo'
d.get('y', "Foo") # 2
e = dict(a = 5, x = 10)
d.update(e) 
# d is now {'a': 5, 'y': 2, 'x': 10}
\end{lstlisting}
